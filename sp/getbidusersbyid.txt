SET @price =(SELECT StartBidPrice FROM vw_biditems WHERE ID = #bidid);SELECT bct.BidItemID, bu.CustomerNo, CONCAT(SUBSTR(c.Firstname,1,1), '.', c.LastName) 'BidUsername', bu.CreatedDate, @price:= CAST(@price + b.BidPrice AS DOUBLE(10,2)) AS price FROM biddingbyusers bu INNER JOIN vw_biduserscount bct ON bct.BidItemID = bu.BidItemID INNER JOIN biddingitems bi ON bi.ID = bu.BidItemID INNER JOIN bidding b ON b.BidCode = bi.BidCode INNER JOIN customer c ON c.CustomerNo = bu.CustomerNo WHERE bu.BidItemID = #bidid ORDER BY bu.CreatedDate DESC
